  kubectl -n {{ .Release.Namespace }} rollout status deploy/{{ include "arizuko.fullname" . }}-gated

  kubectl -n {{ .Release.Namespace }} run -it --rm curl --image=curlimages/curl --restart=Never -- \
    curl -s http://{{ include "arizuko.routerService" . }}:8080/health

to spawn the agent image ({{ .Values.agentImage }}). This requires a docker-runtime
node where that socket exists, and that the arizuko-ant image is present on the node.
If turns hang, check: `kubectl logs deploy/{{ include "arizuko.fullname" . }}-gated`.
{{- else }}

WARNING: agentRuntime.dockerSocket is OFF — gated will start but every agent
turn will FAIL to spawn. Set agentRuntime.dockerSocket=true on a docker-runtime node.
